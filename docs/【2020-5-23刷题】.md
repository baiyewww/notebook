- [x] **leetcode 633 平方数之和**

```cpp
#include <iostream>
using namespace std;
class Solution{
public:
    bool judgeSquareSum(int num){
        if(num == 0) return true;
        if(num < 0) return false;
        int tmp = num;
        int end = sqrt(tmp);
        int start = 0;
        while(start <= end){
            if(start * start == tmp - end * end){
                return true;
            }else
            if(start * start < tmp - end * end){
                start++;
            }else{
                end--;
            }
        }
        return false;
    }
};
```
**注意点**：

1.整数：正数、0、负数，非负数是指0和整数
